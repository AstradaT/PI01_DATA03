CREATE TABLE results (id INTEGER NOT NULL, race_id INTEGER, driver_id INTEGER, constructor_id INTEGER, number INTEGER, grid INTEGER, position INTEGER, position_text VARCHAR, position_order INTEGER, points INTEGER, laps INTEGER, time VARCHAR, milliseconds INTEGER, fastest_lap INTEGER, rank INTEGER, fastest_lap_time FLOAT, fastest_lap_speed FLOAT, status_id INTEGER, PRIMARY KEY (id), FOREIGN KEY(race_id) REFERENCES races (id), FOREIGN KEY(driver_id) REFERENCES drivers (id), FOREIGN KEY(constructor_id) REFERENCES constructors (id));
